# ResponsiveWrapper

## Usage

```jsx
import { ResponsiveWrapper } from '@wordpress/components';

const MyResponsiveWrapper = () => (
	<ResponsiveWrapper
		naturalWidth={ 2000 }
		naturalHeight={ 680 }
	>
		<img src="https://s.w.org/style/images/about/WordPress-logotype-standard.png" alt="WordPress" />
	</ResponsiveWrapper>
);
```
